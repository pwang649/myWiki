Problem 1​

Naive-based Solution​

for (i = 0; i < n; i++) {
    for (j = 0; j < n; j++) {
        for (k = 0; k < n; k++) {
            C[i][j] += A[i][k] * B[k][j];
        }
    }
}

Block-based Solution​

// Block-based
int b = atoi(argv[1]);
int m = n / b;
for (i = 0; i < m; i++) {
    for (j = 0; j < m; j++) {
        for (k = 0; k < m; k++) {
            for (int x = 0; x < b; x++) {
                for (int y = 0; y < b; y++)
                    for (int z = 0; z < b; z++)
                        C[i * b + x][j * b + y] += A[i * b + x][k * b + z] * B[k * b + z][j * b + y];
            }
        }
    }
}
